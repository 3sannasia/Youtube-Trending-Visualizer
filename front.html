SELECT SUM(v.Views),t.Name
FROM Describes d JOIN Videos v ON(d.VideoID=v.VideoID) JOIN Tags t ON (d.TagsID=t.TagsID)
GROUP BY d.TagsID,t.Name
ORDER BY SUM(v.Views) DESC
LIMIT 15;

SELECT v.VideoTitle, c.ChannelName
FROM Videos v JOIN Channels c ON (c.ChannelID = v.ChannelID)
WHERE v.VideoID IN (SELECT VideoID FROM Tags t JOIN Describes d ON (d.TagsID=t.TagsID) WHERE t.Name LIKE "funny") AND v.Likes > 100
ORDER BY v.Views DESC
LIMIT 15;
